PREFIXES =

Regular masks for “prefix”, and they are frozen.

Examples:

[[0, 0, 0, 0],          # 0(first)
[128, 0, 0, 0],         # 1
....,                   # 2 ~ 23
[255, 255, 255, 0],     # 24
....,                   # 24 ~ 31
[255, 255, 255, 255]]   # 32(last)
[[0, 0, 0, 0].freeze].tap { |ret|
  valid_octets = [128, 192, 224, 240, 248, 252, 254, 255].freeze
  base_octets = ret.first.dup

  base_octets.length.times do |n|
    valid_octets.each do |oct|
      mask = base_octets.dup
      mask[n] = oct
      ret << mask.freeze
    end

    base_octets[n] = valid_octets.last
  end

  raise 'must not happen' unless ret.length == 33
}.freeze

def octets_for_integer(int)
  int = int.to_int
  raise RangeError unless (0..4_294_967_295).cover?(int)

  int.to_s(2).rjust(bit_length, '0')
     .scan(/[01]{8}/).map { |s| s.to_i(2) }
end

def parse(str)
  case str
  when /\A#{PATTERN}\z/o
    mask = case
           when $~[:mask]
             octets_for_string($~[:mask])
           when $~[:prefix_length]
             PREFIXES[$~[:prefix_length].to_i]
           else
             FULL_MASK
           end

    new(octets_for_string($~[:octets]), mask)
  else
    raise InvalidAddress, str
  end
end
